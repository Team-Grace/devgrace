import { AB } from '@devgrace/react';

# AB
`AB 테스트`할 때 사용할 수 있는 컴포넌트입니다. 

<br />

## Interface
```tsx
const AB: {
  ({ children }: { children: React.ReactNode }): JSX.Element | null;
  CaseA({ children }: { children: React.ReactNode }): JSX.Element;
  CaseB({ children }: { children: React.ReactNode }): JSX.Element;
}
```

## Example

```tsx
import { AB } from '@devgrace/react';

const { CaseA, CaseB } = AB;

const Example = () => {
  return (
    <AB>
      <CaseA>
        <div>A</div>
      </CaseA>
      <CaseB>
        <div>B</div>
      </CaseB>
    </AB>
  );
}; 
```

## Output

<div>
  <p>페이지를 새로고침 하면 값이 변경됩니다.</p>
  <AB>
    <AB.CaseA>
      <div style={{ fontSize: '2rem'}}>A</div>
    </AB.CaseA>
    <AB.CaseB>
      <div style={{ fontSize: '2rem'}}>B</div>
    </AB.CaseB>
  </AB>
</div>
