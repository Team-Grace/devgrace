# useForceUpdate

A custom hook that forces the component to `re-render` when the returned function is executed.

<br />

## Interface
```tsx
// type DispatchWithoutAction = () => void;
const useForceUpdate: () => React.DispatchWithoutAction
```

## Example

```tsx
import { useForceUpdate } from '@devgrace/react';

const Example = () => {
  const forceUpdate = useForceUpdate();
  
  const handleForceUpdate = useCallback(() => {
    forceUpdate();
  }, [forceUpdate]);

  return (
    <div>
      <button onClick={handleForceUpdate}>Button</button>;
    </div>
  );
};
```
