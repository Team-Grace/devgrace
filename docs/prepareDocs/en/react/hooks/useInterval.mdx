# useInterval

A custom hook to conveniently use `window.setInterval`.

<br />

## Interface
```tsx
const useInterval: (callback: () => void, delay?: number) => void
```

## Example

```tsx
import { useInterval } from '@devgrace/react';

const Example = () => {
 const [count, setCount] = useState(0);

  useInterval(() => {
    setCount(count + 1);
  }, 1000);

  return <h1>{count}</h1>;
};
```
