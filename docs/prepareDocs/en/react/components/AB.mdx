import { AB } from '@devgrace/react';

# AB
Component that can be used when `AB testing`.

Returns `null` if not used correctly, as shown below.

- Using an invalid component type (AB.CaseA. AB.CaseB)
- Using a duplicate component type
- children length is not `2`

<br />

## Interface
```tsx
const AB: {
  ({ children }: { children: React.ReactNode }): JSX.Element | null;
  CaseA({ children }: { children: React.ReactNode }): JSX.Element;
  CaseB({ children }: { children: React.ReactNode }): JSX.Element;
}
```

## Example

```tsx
import { AB } from '@devgrace/react';

const { CaseA, CaseB } = AB;

const Example = () => {
  return (
    <AB>
      <CaseA>
        <div>A</div>
      </CaseA>
      <CaseB>
        <div>B</div>
      </CaseB>
    </AB>
  );
}; 
```

## Output

<div>
  <p>Refreshing the page will change the value.</p>
  <AB>
    <AB.CaseA>
      <div style={{ width: "50px", backgroundColor: "green", color: "white", fontSize: '2rem', textAlign: 'center'}}>A</div>
    </AB.CaseA>
    <AB.CaseB>
      <div style={{width: "50px", backgroundColor: "blue", color: "white", fontSize: '2rem', textAlign: 'center'}}>B</div>
    </AB.CaseB>
  </AB>
</div>
