import { AB } from '@devgrace/react';

# AB
Component that can be used when `AB testing`.

Returns `null` if not used correctly, as shown below.

- Using an invalid component type (Available components: AB.CaseA. AB.CaseB)
- Using a duplicate component type
- children length is not `2`

<br />

## Interface
```tsx
interface ABProps {
  onAction?: () => void;
}

const AB: {
  ({ children }: PropsWithChildren): JSX.Element | null;
  CaseA({ children, onAction }: PropsWithChildren<ABProps>): JSX.Element;
  CaseB({ children, onAction }: PropsWithChildren<ABProps>): JSX.Element;
};
```

## Usage

```tsx
import { AB } from '@devgrace/react';

const { CaseA, CaseB } = AB;

const Example = () => {
  return (
    <AB>
      <CaseA onAction={() => console.log("CaseA Renader")}>
        <div>A</div>
      </CaseA>
      <CaseB onAction={() => console.log("CaseB Renader")}>
        <div>B</div>
      </CaseB>
    </AB>
  );
}; 
```
