@devgrace/ui Theme Colors


Usage​

import { colors } from '@devgrace/ui'

const Example = () => {
  return (
    <>
      <div style={{ width: "100%", height: '50px', backgroundColor: colors.red[500] }} />
      <div style={{ width: "100%", height: '50px', backgroundColor: colors.yellow[500] }} />
      <div style={{ width: "100%", height: '50px', backgroundColor: colors.indigo[500] }} />
    </>
  )
}
